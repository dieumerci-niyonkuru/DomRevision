// Create a table element const table = document.createElement('table'); // Create a header row const headerRow = document.createElement('tr'); const headers = ['Name', 'Age', 'City']; headers.forEach(headerText => { const header = document.createElement('th'); header.textContent = headerText; headerRow.appendChild(header); }); table.appendChild(headerRow); // Create some data rows const data = [ { name: 'Alice', age: 25, city: 'New York' }, { name: 'Bob', age: 30, city: 'Los Angeles' }, { name: 'Charlie', age: 35, city: 'Chicago' } ]; data.forEach(item => { const row = document.createElement('tr'); const nameCell = document.createElement('td'); nameCell.textContent = item.name; row.appendChild(nameCell); const ageCell = document.createElement('td'); ageCell.textContent = item.age; row.appendChild(ageCell); const cityCell = document.createElement('td'); cityCell.textContent = item.city; row.appendChild(cityCell); table.appendChild(row); }); // Append the table to the body or a specific element document.body.appendChild(table); // Create buttons const changeColorButton = document.createElement('button'); changeColorButton.textContent = 'Change Color'; changeColorButton.onclick = () => { table.style.backgroundColor = table.style.backgroundColor === 'lightblue' ? '' : 'lightblue'; }; const changeFontButton = document.createElement('button'); changeFontButton.textContent = 'Change Font Size'; changeFontButton.onclick = () => { const currentSize = window.getComputedStyle(table).fontSize; table.style.fontSize = currentSize === '16px' ? '20px' : '16px'; }; const scrollButton = document.createElement('button'); scrollButton.textContent = 'Scroll Left/Right'; scrollButton.onclick = () => { table.scrollLeft += 100; // Scrolls 100 pixels to the right }; // Append buttons to the body document.body.appendChild(changeColorButton); document.body.appendChild(changeFontButton); document.body.appendChild(scrollButton); // Style the table for better visibility table.style.borderCollapse = 'collapse'; table.style.width = '100%'; table.style.overflowX = 'auto'; table.style.whiteSpace = 'nowrap'; table.querySelectorAll('th, td').forEach(cell => { cell.style.border = '1px solid black'; cell.style.padding = '8px'; });
